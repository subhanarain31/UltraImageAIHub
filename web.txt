<script>
async function generateImage() {
    const prompt = document.getElementById("prompt").value;
    const resultDiv = document.getElementById("result");
    resultDiv.innerHTML = "⏳ Generating image...";

    // ⚠️ SAFE WAY: Token should NOT be in public repo
    // For local testing, you can add token here temporarily
    const HUGGINGFACE_API_KEY = prompt("Enter your HuggingFace token (local testing only)");

    const response = await fetch(
        "https://api-inference.huggingface.co/models/prompthero/openjourney",
        {
            method: "POST",
            headers: { 
                "Authorization": `Bearer ${HUGGINGFACE_API_KEY}`,
                "Content-Type": "application/json"
            },
            body: JSON.stringify({ inputs: prompt })
        }
    );

    const blob = await response.blob();
    const imageURL = URL.createObjectURL(blob);
    resultDiv.innerHTML = `<img src="${imageURL}">`;
}
</script>
